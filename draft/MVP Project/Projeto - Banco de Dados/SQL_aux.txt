Este arquivo é destinado para o registro de criação do diagrama no programa draw.io a partir de um código sql.

create table User(
user_id,
username,
user_email,
user_password,
);

create table user_metadata(
user_id,
full_name,
cpf,
rg,
photo_profile_url,
birth_date,
registered_at,
last_login,
last_purchase,
created_by,
updated_at     
);

create table order(
order_id,
user_id,
total_amout,
item_count,
order_date,
payment_method
);

CREATE TABLE products (
product_id,
user_id,
name,
cost_price,
sale_price,
stock_quantity,
minimum_stock,
maximum_stock,
category
);

CREATE TABLE product_metadata (
product_id,
ad_description,
slogan,
technical_details,
brand,
product_images_id,
registered_at,
updated_at
);

CREATE TABLE product_image(
product_images_id,
product_image_url
);

create table order_items(
order_id,
product_id,
price,
count
);

create table feedback(
feedback_id,
entity_id,
type_feedback,
rating,
comment,

dislikes_count,

create table feedback_comments(
comment_id,
